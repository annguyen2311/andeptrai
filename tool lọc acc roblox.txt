import tkinter as tk
from tkinter import filedialog, messagebox

def split_data(input_file, output_file1, output_file2):
    try:
        with open(input_file, 'r', encoding='utf-8') as file:
            data_list = file.readlines()

        accounts = []
        cookies = []

        for data in data_list:
            data = data.strip()
            if ":" in data:
                parts = data.split(':', 2)
                account_part = parts[0] + ':' + parts[1]
                accounts.append(account_part.rstrip(':'))
                if '_|' in data:
                    cookies.append('_|' + data.split('_|', 1)[1])

        with open(output_file1, 'w', encoding='utf-8') as file1:
            for item in accounts:
                file1.write(item + '\n')

        with open(output_file2, 'w', encoding='utf-8') as file2:
            for item in cookies:
                file2.write(item + '\n')

        messagebox.showinfo("Hoàn thành", "Quá trình tách dữ liệu đã hoàn tất.")
    
    except FileNotFoundError:
        messagebox.showerror("Lỗi", "File đầu vào không tồn tại. Vui lòng kiểm tra lại đường dẫn.")
    except Exception as e:
        messagebox.showerror("Lỗi", f"Đã xảy ra lỗi: {str(e)}")

def select_input_file():
    file_path = filedialog.askopenfilename(title="Chọn file đầu vào")
    input_entry.delete(0, tk.END)
    input_entry.insert(0, file_path)

def select_output_file1():
    file_path = filedialog.asksaveasfilename(title="Chọn file đầu ra 1", defaultextension=".txt", filetypes=[("Text files", "*.txt")])
    output_entry1.delete(0, tk.END)
    output_entry1.insert(0, file_path)

def select_output_file2():
    file_path = filedialog.asksaveasfilename(title="Chọn file đầu ra 2", defaultextension=".txt", filetypes=[("Text files", "*.txt")])
    output_entry2.delete(0, tk.END)
    output_entry2.insert(0, file_path)

def start_splitting():
    input_file = input_entry.get()
    output_file1 = output_entry1.get()
    output_file2 = output_entry2.get()
    if input_file and output_file1 and output_file2:
        split_data(input_file, output_file1, output_file2)
    else:
        messagebox.showwarning("Cảnh báo", "Vui lòng chọn tất cả các file cần thiết.")

# Tạo cửa sổ chính
root = tk.Tk()
root.title("Công Cụ Tách Dữ Liệu")

# Label và Entry cho file đầu vào
input_label = tk.Label(root, text="File đầu vào:")
input_label.grid(row=0, column=0, padx=10, pady=10)
input_entry = tk.Entry(root, width=50)
input_entry.grid(row=0, column=1, padx=10, pady=10)
input_button = tk.Button(root, text="Chọn file...", command=select_input_file)
input_button.grid(row=0, column=2, padx=10, pady=10)

# Label và Entry cho file đầu ra 1
output_label1 = tk.Label(root, text="File đầu ra 1:")
output_label1.grid(row=1, column=0, padx=10, pady=10)
output_entry1 = tk.Entry(root, width=50)
output_entry1.grid(row=1, column=1, padx=10, pady=10)
output_button1 = tk.Button(root, text="Chọn file...", command=select_output_file1)
output_button1.grid(row=1, column=2, padx=10, pady=10)

# Label và Entry cho file đầu ra 2
output_label2 = tk.Label(root, text="File đầu ra 2:")
output_label2.grid(row=2, column=0, padx=10, pady=10)
output_entry2 = tk.Entry(root, width=50)
output_entry2.grid(row=2, column=1, padx=10, pady=10)
output_button2 = tk.Button(root, text="Chọn file...", command=select_output_file2)
output_button2.grid(row=2, column=2, padx=10, pady=10)

# Nút bắt đầu tách
start_button = tk.Button(root, text="Bắt đầu tách", command=start_splitting)
start_button.grid(row=3, column=1, padx=10, pady=20)

# Khởi chạy giao diện
root.mainloop()
